"Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb Output: a=2 b=2 c=2 d=1 e=2 f=1"

                    let word = "abcadeecfb";
let count = new Map();

for(let i=0; i < word.length; i++)
{
    if(count.has(word[i]))
    {
        count.set(word[i],Number(count.get(word[i]))+1);
    }
    else
    {
        count.set(word[i],1);
    }
}

count.forEach((value,key)=>
{
    console.log(key + " = " + value);
});
